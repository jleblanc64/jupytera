docker build -t my-hello-app .

# 1. Set variables (Optional, but recommended for simplicity)
SERVICE_NAME="helloworld-python"
REPO_NAME="helloworld-python-repo"
REGION="us-central1" # Recommended for free tier
PROJECT_ID=$(gcloud config get-value project)
IMAGE_NAME="${REGION}-docker.pkg.dev/${PROJECT_ID}/${SERVICE_NAME}-repo/${SERVICE_NAME}:latest"

# Command to create a Docker repository
gcloud artifacts repositories create $REPO_NAME \
  --repository-format=docker \
  --location=$REGION \
  --description="Repo for helloworld python images"

# 2. Configure Docker authentication for Artifact Registry
gcloud auth configure-docker "${REGION}-docker.pkg.dev"

# 3. Tag your local image (assuming you have a local image named 'my-hello-app')
docker tag my-hello-app:latest $IMAGE_NAME

docker push $IMAGE_NAME
