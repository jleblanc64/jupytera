SERVICE_NAME="helloworld-python"
REPO_NAME="helloworld-python-repo"
REGION="us-central1" # Recommended for free tier
PROJECT_ID=$(gcloud config get-value project)
IMAGE_NAME="${REGION}-docker.pkg.dev/${PROJECT_ID}/${SERVICE_NAME}-repo/${SERVICE_NAME}:latest"

gcloud services enable run.googleapis.com

gcloud run deploy $SERVICE_NAME \
  --image $IMAGE_NAME \
  --region $REGION \
  --platform managed \
  --cpu 0.25 \
  --memory 256Mi \
  --min-instances 0 \
  --allow-unauthenticated # Makes the service publicly accessible

# Note: The command will output the final Service URL.
